  <script>
    const searchInput = document.getElementById("searchInput");
    const counter = document.getElementById("counter");
    const noResults = document.getElementById("noResults");
    const traitCards = [...document.querySelectorAll(".trait-card")];

    function filterTraits() {
      const keyword = searchInput.value.trim().toLowerCase();
      let visibleCount = 0;

      traitCards.forEach(card => {
        const isMatch = card.textContent.toLowerCase().includes(keyword);
        card.style.display = isMatch ? "block" : "none";
        if (isMatch) visibleCount += 1;
      });

      noResults.classList.toggle("show", visibleCount === 0);
      counter.textContent = `${visibleCount} of ${traitCards.length} traits shown`;
    }

    searchInput.addEventListener("input", filterTraits);
    filterTraits();
  </script>
